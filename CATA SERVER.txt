<script>
(async function() {
  let allProducts = [];
  let imageMap = {};
  let lastUpdate = null;

  const grid = document.getElementById('productGrid');
  const searchInput = document.getElementById('searchInput');
  const syncBadge = document.getElementById('syncBadge');
  const syncText = document.getElementById('syncText');
  const productCount = document.getElementById('productCount');

  // Carica catalogo dal server
  async function loadCatalog() {
    try {
      const resp = await fetch('/api/catalogo?' + Date.now());
      if (!resp.ok) throw new Error('Server non raggiungibile');
      
      const data = await resp.json();
      allProducts = data.products || [];
      imageMap = data.imageMap || {};
      lastUpdate = data.lastUpdate;
      
      if (lastUpdate) {
        syncBadge.style.display = 'inline-flex';
        syncText.textContent = 'Sincronizzato: ' + new Date(lastUpdate).toLocaleString('it-IT');
      }
      
      renderProducts();
    } catch(e) {
      console.error('Errore caricamento:', e);
      grid.innerHTML = '<div class="no-results"><i class="fas fa-exclamation-triangle"></i><p>Impossibile caricare il catalogo</p><p style="font-size:0.8rem;">Assicurati che il server sia attivo</p></div>';
      productCount.textContent = '0 prodotti';
    }
  }

  function renderProducts() {
    const term = searchInput.value.toLowerCase().trim();
    
    const filtered = allProducts.filter(p => {
      if (!term) return true;
      return (p.name || '').toLowerCase().includes(term) || 
             (p.code || '').toLowerCase().includes(term) ||
             (p.category || '').toLowerCase().includes(term);
    });
    
    productCount.textContent = filtered.length + ' prodotti' + 
      (allProducts.length !== filtered.length ? ' (su ' + allProducts.length + ' totali)' : '');
    
    if (filtered.length === 0) {
      grid.innerHTML = '<div class="no-results"><i class="fas fa-box-open" style="font-size:3rem;opacity:0.3;display:block;margin-bottom:1rem;"></i>Nessun prodotto trovato</div>';
      return;
    }
    
    grid.innerHTML = filtered.map(p => {
      const img = imageMap[p.code] || p.img || getPlaceholder();
      const isDanea = p.source === 'danea';
      
      return `
        <div class="card${isDanea ? ' danea' : ''}">
          <div class="img-wrapper">
            <img src="${img}" alt="${p.name || ''}" loading="lazy" onerror="this.src='${getPlaceholder()}'">
          </div>
          <span class="product-code">
            ${p.code || ''}
            ${isDanea ? '<span class="badge badge-danea">DANEA L1</span>' : ''}
          </span>
          <h3>${p.name || 'N/D'}</h3>
          <div class="category"><i class="fas fa-tag"></i> ${p.category || 'N/D'}</div>
          <div class="manufacturer"><i class="fas fa-industry"></i> ${p.manufacturer || 'N/D'}</div>
          ${p.note ? '<div class="note">' + p.note + '</div>' : ''}
          <div class="price">€ ${(p.price || '0,00').replace('.', ',')} <small>+IVA</small></div>
        </div>`;
    }).join('');
  }

  function getPlaceholder() {
    return 'data:image/svg+xml,' + encodeURIComponent(
      '<svg xmlns="http://www.w3.org/2000/svg" width="200" height="200" viewBox="0 0 200 200">' +
      '<rect fill="#f0fdf4" width="200" height="200" rx="20"/>' +
      '<text fill="#065f46" x="100" y="100" text-anchor="middle" font-size="16" font-family="sans-serif" font-weight="bold">DANEA</text>' +
      '</svg>'
    );
  }

  // Ricerca in tempo reale
  searchInput.addEventListener('input', renderProducts);

  // Carica all'avvio
  await loadCatalog();

  // Aggiorna ogni 30 secondi
  setInterval(loadCatalog, 30000);

  console.log('?? Catalogo Elettrico pronto');
  console.log('?? Listino Danea: 1 (NetPrice1)');
  console.log('?? Auto-aggiornamento: ogni 30 secondi');
})();
</script>
